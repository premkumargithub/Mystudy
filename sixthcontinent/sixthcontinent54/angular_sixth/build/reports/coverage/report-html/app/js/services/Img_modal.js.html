app.service('ImgCommentService', function ($http, $timeout) {
    return {
        getMediaInfo : function(opt, callback) { 
            var url = APP.service.getComment+"?access_token="+APP.accessToken;
            doPost($http, url, opt, function(data) {
                callback(data);
            });
        },

            var url = APP.service.addComment+"?access_token="+APP.accessToken;

    }
